def print_num_balls_by_color(inches: float, dollars: float, cost_dict: dict):
    print(f"\nThe number of {2 * inches:4.2f}-inch diameter balls "
          f"you can buy with ${dollars:4.2f} dollar(s) by color is:")
    for color, cost in cost_dict.items():
        num_balls = int(dollars / cost)
        print(f"{color:<6} {num_balls}")
